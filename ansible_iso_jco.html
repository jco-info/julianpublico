Playbook para clientes LDAP

6.1Crear site-clients.yml
bash
nano site-clients.yml
site-clients.yml
---
- name: Configurar clientes LDAP
  hosts: compute-0-1, compute-0-2, compute-0-3
  become: yes
  tasks:
    - name: Instalar paquetes cliente LDAP
      apt:
        name:
          - libnss-ldap
          - libpam-ldap
          - ldap-utils
          - nscd
        state: present
        update_cache: yes

    - name: Configurar /etc/ldap/ldap.conf
      copy:
        dest: /etc/ldap/ldap.conf
        content: |
          BASE    {{ ldap_base_dn }}
          URI     ldap://{{ gateway_ip }}

    - name: Configurar /etc/ldap.conf
      copy:
        dest: /etc/ldap.conf
        content: |
          base {{ ldap_base_dn }}
          uri ldap://{{ gateway_ip }}
          ldap_version 3
          rootbinddn cn=admin,{{ ldap_base_dn }}

    - name: Crear archivo de credenciales root
      copy:
        dest: /etc/ldap.secret
        content: "{{ ldap_admin_password }}\n"
        mode: '0600'

    - name: Configurar /etc/nsswitch.conf para LDAP
      lineinfile:
        path: /etc/nsswitch.conf
        regexp: "^{{ item.key }}:"
        line: "{{ item.key }}: files ldap"
        backup: yes
      loop:
        - { key: passwd }
        - { key: group }
        - { key: shadow }

    - name: Configurar PAM para autenticación LDAP
      shell: |
        debconf-set-selections <

Fase 7

Playbook para servidor NFS

7.1Crear site-nfs.yml
bash
nano site-nfs.yml
site-nfs.yml
---
- name: Configurar servidor NFS
  hosts: compute-0-0
  become: yes
  tasks:
    - name: Instalar servidor NFS
      apt:
        name: nfs-kernel-server
        state: present
        update_cache: yes

    - name: Crear directorios home para usuarios LDAP
      file:
        path: "/home/{{ item.uid }}"
        state: directory
        owner: "{{ item.uid_number }}"
        group: "{{ ldap_group_gid }}"
        mode: '0755'
      loop: "{{ ldap_users }}"

    - name: Crear archivo de bienvenida
      copy:
        dest: "/home/{{ item.uid }}/bienvenida.txt"
        content: "Bienvenido {{ item.uid }}\n"
        owner: "{{ item.uid_number }}"
        group: "{{ ldap_group_gid }}"
        mode: '0644'
      loop: "{{ ldap_users }}"

    - name: Configurar /etc/exports
      copy:
        dest: /etc/exports
        content: |
          /home           {{ internal_network }}(rw,sync,no_subtree_check,no_root_squash)

    - name: Aplicar configuración NFS
      command: exportfs -a
      changed_when: true

    - name: Reiniciar servidor NFS
      systemd:
        name: nfs-kernel-server
        state: restarted
        enabled: yes
bash
ansible-playbook site-nfs.yml -u root
Fase 8

Playbook para Autofs en clientes

8.1Crear site-autofs.yml
bash
nano site-autofs.yml
site-autofs.yml
---
- name: Configurar autofs en clientes
  hosts: compute-0-1, compute-0-2, compute-0-3
  become: yes
  tasks:
    - name: Instalar autofs
      apt:
        name: autofs
        state: present

    - name: Configurar /etc/auto.master
      lineinfile:
        path: /etc/auto.master
        line: "/home /etc/auto.home --timeout=60"
        insertafter: EOF
        state: present
        backup: yes

    - name: Crear archivo /etc/auto.home
      copy:
        dest: /etc/auto.home
        content: |
          * -fstype=nfs,rw,hard,intr,noatime,nolock {{ gateway_ip }}:/home/&
        mode: '0644'

    - name: Reiniciar autofs
      systemd:
        name: autofs
        state: restarted
        enabled: yes
bash
ansible-playbook site-autofs.yml -u root

Fase 10

Playbook para Munge (todos los nodos)

10.1Crear site-munge.yml
bash
nano site-munge.yml
site-munge.yml
---
- name: Configurar Munge en todos los nodos
  hosts: all
  become: yes
  tasks:
    - name: Instalar munge
      apt:
        name: munge
        state: present
        update_cache: yes

    - name: Asegurar que el servicio munge está detenido para configurar
      systemd:
        name: munge
        state: stopped
      when: inventory_hostname != "compute-0-0"

    - name: Generar clave munge en el servidor
      command: /usr/sbin/create-munge-key
      when: inventory_hostname == "compute-0-0"
      args:
        creates: /etc/munge/munge.key

    - name: Copiar clave munge desde el servidor a los clientes
      synchronize:
        src: /etc/munge/munge.key
        dest: /etc/munge/munge.key
        mode: push
      delegate_to: compute-0-0
      when: inventory_hostname != "compute-0-0"

    - name: Configurar permisos de la clave
      file:
        path: /etc/munge/munge.key
        owner: munge
        group: munge
        mode: '0400'

    - name: Habilitar y arrancar munge
      systemd:
        name: munge
        state: started
        enabled: yes

    - name: Verificar que munge está funcionando
      shell: munge -n | unmunge
      register: munge_test
      changed_when: false
      failed_when: "'Success' not in munge_test.stdout"
bash
ansible-playbook site-munge.yml -u root

12.1Crear site-slurm-controller.yml
bash
nano site-slurm-controller.yml
site-slurm-controller.yml
---
- name: Configurar controlador Slurm
  hosts: compute-0-0
  become: yes
  tasks:
    - name: Instalar slurm-wlm
      apt:
        name: slurm-wlm
        state: present
        update_cache: yes

    - name: Crear directorios necesarios
      file:
        path: "{{ item }}"
        state: directory
        owner: slurm
        group: slurm
        mode: '0755'
      loop:
        - /var/spool/slurmctld
        - /var/log/slurm

        src: templates/slurm.conf.j2

        group: slurm
        mode: '0644'

        name: slurmctld
        state: started
        enabled: yes
