📘 SCRIPT COMPLETO · TABLA EMPLE 
-- =====================================================
-- BASE DE DATOS PARA PRÁCTICAS DE FUNCIONES DE FECHA
-- ASIGNATURA: BASES DE DATOS - ASIR
-- TABLA: EMPLE (Empleados de una empresa)
-- =====================================================

-- PASO 1: CREAR LA TABLA
-- =====================================================
CREATE TABLE emple (
    id_empleado    NUMBER(4) PRIMARY KEY,
    apellido       VARCHAR2(30) NOT NULL,
    nombre         VARCHAR2(30),
    oficio         VARCHAR2(20),
    fecha_alt      DATE,        -- Fecha de alta/contratación
    salario        NUMBER(8,2),
    comision       NUMBER(8,2),
    dept_no        NUMBER(2)
);

🖥️ CONFIGURACIÓN VISUAL (SQL*Plus)

-- Formato de columnas

COLUMN apellido FORMAT A15
COLUMN nombre FORMAT A12
COLUMN oficio FORMAT A12
COLUMN fecha_alt FORMAT A12
COLUMN salario FORMAT 999,999
COLUMN comision FORMAT 999,999
COLUMN dept_no FORMAT 99

-- Formato de fecha legible
ALTER SESSION SET NLS_DATE_FORMAT = 'DD/MM/YYYY';

-- Ver la tabla ordenada
SELECT id_empleado, apellido, nombre, oficio, fecha_alt, salario, dept_no 
FROM emple 
ORDER BY fecha_alt;
📌 EJERCICIOS PRÁCTICOS
EJERCICIO 1 Sumar 15 horas a la fecha de hoy.
SELECT SYSDATE + 15/24 FROM DUAL;
EJERCICIO 2 Mostrar el texto del día en el cual naciste. ¿En qué día naciste? (cambia la fecha)
SELECT TO_CHAR(TO_DATE('06/10/1967','DD/MM/YYYY'), 'DAY') FROM DUAL;
EJERCICIO 4 ¿Cuántos días tiene el mes de febrero del año que viene? Consulta válida independientemente del año actual.
SELECT ADD_MONTHS(TRUNC(SYSDATE + 365, 'YEAR'), 2) - ADD_MONTHS(TRUNC(SYSDATE + 365, 'YEAR'), 1) AS dias_febrero FROM DUAL;
-- Alternativa más clara: SELECT LAST_DAY(TO_DATE('01/02/' || EXTRACT(YEAR FROM SYSDATE+365), 'DD/MM/YYYY')) FROM DUAL;
EJERCICIO 5 Muestra de cada empleado su apellido y el nombre del mes en que llegó a la empresa.
SELECT apellido, TO_CHAR(fecha_alt, 'Month') AS mes_contratacion FROM emple;
EJERCICIO 6 Visualiza el apellido de cada empleado y el número de meses (sin decimales) que lleva trabajando en la empresa.
SELECT apellido, ROUND(MONTHS_BETWEEN(SYSDATE, fecha_alt)) AS meses_trabajados FROM emple;
EJERCICIO 7 ¿Cuántos meses lleva trabajando el empleado que más meses lleva?
SELECT MAX(ROUND(MONTHS_BETWEEN(SYSDATE, fecha_alt))) AS max_antiguedad_meses FROM emple;
➕ EXTRA Muestra el nombre completo del mes de contratación para cada empleado (ej: 'Enero', 'Febrero')
SELECT apellido, TO_CHAR(fecha_alt, 'Month', 'NLS_DATE_LANGUAGE=SPANISH') FROM emple;
